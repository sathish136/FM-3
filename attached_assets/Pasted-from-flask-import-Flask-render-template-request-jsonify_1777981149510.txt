def get_encryption_key():
    """Generate encryption key from secret key"""
    kdf = PBKDF2HMAC(
        algorithm=hashes.SHA256(),
        length=32,
        salt=b'wtt_erp_salt',  # Use a proper salt in production
        iterations=100000,
    )
    key = base64.urlsafe_b64encode(kdf.derive(app.secret_key.encode()))
    return key

def encrypt_password(password):
    """Encrypt password for storage"""
    f = Fernet(get_encryption_key())
    return f.encrypt(password.encode()).decode()

def decrypt_password(encrypted_password):
    """Decrypt password from storage"""
    try:
        f = Fernet(get_encryption_key())
        return f.decrypt(encrypted_password.encode()).decode()
    except:
        return ''

def login_required(f):
    @wraps(f)
    def decorated_function(*args, **kwargs):
        if 'user_data' not in session:
            return jsonify({'error': 'Authentication required'}), 401
        return f(*args, **kwargs)
    return decorated_function

def permission_required(permission_key):
    def decorator(f):
        @wraps(f)
        def decorated_function(*args, **kwargs):
            if 'user_data' not in session:
                return jsonify({'error': 'Authentication required'}), 401
            
            user_data = session['user_data']
            permissions = user_data.get('message_data', {})
            
            # Check if permission is granted (value == '1')
            if permissions.get(permission_key) != '1':
                return jsonify({'error': f'Permission denied: {permission_key}'}), 403
            
            return f(*args, **kwargs)
        return decorated_function
    return decorator

class ERPNextAPI:
    def __init__(self, base_url):
        self.base_url = base_url
    
    def login(self, username, password):
        """Authenticate with ERPNext API and get token"""
        url = f"{self.base_url}{ERP_LOGIN_ENDPOINT}"
        headers = {
            'Content-Type': 'application/json',
            'Accept': 'application/json'
        }
        data = {
            "username": username,
            "password": password
        }
        
        try:
            response = requests.post(url, headers=headers, json=data, timeout=30)
            if response.status_code == 200:
                result = response.json()
                # Parse the message string to dictionary
                if 'message' in result and result['message']:
                    try:
                        # Clean and parse the message string
                        message_str = result['message'].replace("'", '"')
                        message_data = json.loads(message_str)
                        result['message_data'] = message_data
                        
                        # Extract token from message_data - token already has "Token" prefix
                        token = message_data.get('token', '')
                        if token:
                            result['token'] = token  # Store as-is, don't modify
                    except json.JSONDecodeError as e:
                        print(f"Error parsing message data: {e}")
                        # Try alternative parsing
                        try:
                            # Remove curly braces and split by comma
                            message_str = result['message'].strip("{}").replace("'", "")
                            message_data = {}
                            for item in message_str.split(', '):
                                if ': ' in item:
                                    key, value = item.split(': ', 1)
                                    message_data[key.strip()] = value.strip()
                            result['message_data'] = message_data
                            result['token'] = message_data.get('token', '')
                        except Exception as e2:
                            print(f"Alternative parsing also failed: {e2}")
                            result['message_data'] = {}
                            result['token'] = ''
                    except Exception as e:
                        print(f"Unexpected error parsing message: {e}")
                        result['message_data'] = {}
                        result['token'] = ''
                else:
                    result['message_data'] = {}
                    result['token'] = ''
                return result
            return {"error": f"API Error: {response.status_code}", "details": response.text}
        except requests.exceptions.Timeout:
            return {"error": "ERPNext API timeout. Please try again."}
        except requests.exceptions.ConnectionError:
            return {"error": "Cannot connect to ERPNext server. Please check network."}
        except Exception as e:
            return {"error": str(e)}

# Location service class
class LocationService:
    def __init__(self):
        self.geoip_reader = None
        try:
            if os.path.exists(GEOIP_DATABASE_PATH):
                self.geoip_reader = geoip2.database.Reader(GEOIP_DATABASE_PATH)
                print("GeoIP database loaded successfully")
            else:
                print(f"GeoIP database not found at {GEOIP_DATABASE_PATH}")
        except Exception as e:
            print(f"Error loading GeoIP database: {e}")
    
    def get_ip_location(self, ip_address):
        """Get location from IP address using GeoIP"""
        if not self.geoip_reader:
            return None
        
        try:
            response = self.geoip_reader.city(ip_address)
            return {
                'lat': response.location.latitude,
                'log': response.location.longitude,
                'city': response.city.name,
                'country': response.country.name,
                'postal_code': response.postal.code,
                'source': 'geoip'
            }
        except Exception as e:
            print(f"Error getting IP location: {e}")
            return None
    
    def get_location_from_ipinfo(self, ip_address):
        """Get location from ipinfo.io service"""
        try:
            response = requests.get(f"https://ipinfo.io/{ip_address}/json", timeout=5)
            if response.status_code == 200:
                data = response.json()
                if 'loc' in data:
                    lat, log = data['loc'].split(',')
                    return {
                        'lat': float(lat),
                        'log': float(log),
                        'city': data.get('city', ''),
                        'region': data.get('region', ''),
                        'country': data.get('country', ''),
                        'postal': data.get('postal', ''),
                        'org': data.get('org', ''),
                        'source': 'ipinfo'
                    }
        except Exception as e:
            print(f"Error getting location from ipinfo: {e}")
        return None
    
    def reverse_geocode(self, lat, log):
        """Get address from coordinates using Nominatim"""
        try:
            response = requests.get(
                f"https://nominatim.openstreetmap.org/reverse",
                params={
                    'lat': lat,
                    'lon': log,
                    'format': 'json',
                    'zoom': 18
                },
                headers={'User-Agent': 'WTT-ERP-System/1.0'},
                timeout=10
            )
            if response.status_code == 200:
                data = response.json()
                address = data.get('display_name', '')
                return {
                    'address': address,
                    'full_data': data
                }
        except Exception as e:
            print(f"Error reverse geocoding: {e}")
        return None
    
    def get_client_ip(self):
        """Get client IP address from request"""
        if request.headers.get('X-Forwarded-For'):
            ip = request.headers.get('X-Forwarded-For').split(',')[0]
        elif request.headers.get('X-Real-IP'):
            ip = request.headers.get('X-Real-IP')
        else:
            ip = request.remote_addr
        
        # Handle localhost
        if ip in ['127.0.0.1', 'localhost']:
            # Try to get external IP
            try:
                response = requests.get('https://api.ipify.org?format=json', timeout=5)
                if response.status_code == 200:
                    ip_data = response.json()
                    ip = ip_data.get('ip', ip)
            except:
                pass
        
        return ip

        }

    }
    
    # Request body
    payload = {
        'month': month,
        'year': year
    }
    
    try:
        print(f"ðŸ“Š Fetching attendance for {month} {year}")
        print(f"ðŸ”‘ Using token: {token[:20]}...")
        
        # Make request to ERPNext
        response = requests.post(url, headers=headers, json=payload, timeout=30)

            result = response.json()

@login_required
def get_recent_activities():
    """Get recent activities from check-in data"""
    user_data = session['user_data']
    
    # Get check-in data from ERPNext
    activities = []
    try:
        if 'token' in user_data and user_data['token']:
            response = erp_api.get_checkin(user_data['username'], user_data['token'])
            if 'error' not in response:
                checkin_data = response.get('message', []) if isinstance(response, dict) else response
                
                for record in checkin_data[:5]:
                    if isinstance(record, dict):
                        record_time = record.get('time', '')
                        log_type = record.get('log', '')
                        
                        # Format time for display
                        try:
                            record_datetime = datetime.strptime(record_time, '%Y-%m-%d %H:%M:%S')
                            now = datetime.now()
                            
                            if record_datetime.date() == now.date():
                                display_time = 'Today'
                            elif record_datetime.date() == (now - timedelta(days=1)).date():
                                display_time = 'Yesterday'
                            else:
                                days_diff = (now.date() - record_datetime.date()).days
                                if days_diff < 7:
                                    display_time = f'{days_diff} days ago'
                                else:
                                    display_time = record_datetime.strftime('%d %b')
                        except:
                            display_time = record_time
                        
                        activity_type = 'checkin' if log_type == 'IN' else 'checkout'
                        details = f"{'Checked in' if log_type == 'IN' else 'Checked out'} at {record_time}"
                        
                        activities.append({
                            'id': len(activities) + 1,
                            'type': activity_type,
                            'time': display_time,
                            'details': details
                        })
    except Exception as e:
        print(f"Error fetching recent activities: {e}")
    
    # If no check-in data, add sample activities
    if not activities:
        activities = [
            {
                'id': 1,
                'type': 'system',
                'time': 'Today',
                'details': 'Welcome to WTT ERP System'
            },
            {
                'id': 2,
                'type': 'info',
                'time': 'Just now',
                'details': 'You have successfully logged in'
            }
        ]
    
    return jsonify({'activities': activities})

@app.route('/api/leave/apply', methods=['POST'])
@login_required
def apply_leave():
    """Submit leave application to ERPNext"""
    try:
        user_data = session.get('user_data')
        if not user_data:
            return jsonify({'success': False, 'error': 'No user data in session'}), 401
        
        data = request.get_json()
        if not data:
            return jsonify({'success': False, 'error': 'No data provided'}), 400
        
        token = user_data.get('token', '')
        if not token:
            return jsonify({'success': False, 'error': 'No authentication token'}), 401
        
        print(f"ðŸ“ Applying leave for {data['emp']}: {data['from_date']} to {data['to_date']}")
        
        url = "https://erp.wttint.com/api/method/wtt_module.customization.custom.mobile_web.create_leave_new"
        
        headers = {
            'Authorization': token,
            'Content-Type': 'application/json'
        }
        
        payload = {
            "emp": data['emp'],
            "from_date": data['from_date'],
            "to_date": data['to_date'],
            "day": data['day'],
            "leave_type": data['leave_type'],
            "reason": data['reason']
        }
        
        response = requests.post(url, headers=headers, json=payload, timeout=30)
        
        print(f"ðŸ“¡ ERPNext response status: {response.status_code}")
        
        if response.status_code == 200:
            result = response.json()
            print(f"ðŸ“¥ ERPNext response: {result}")
            
            # Check various success formats
            message = result.get('message', '')
            
            # If message is "Success" or contains "success", consider it successful
            if message == 'Success' or message == 'success' or 'success' in str(message).lower():
                return jsonify({
                    'success': True,
                    'message': 'Leave application submitted successfully',
                    'data': result
                })
            else:
                # If we got 200 but message is not "Success", still consider it success
                # as ERPNext sometimes returns different success messages
                return jsonify({
                    'success': True,
                    'message': message or 'Application submitted',
                    'data': result
                })
        else:
            error_text = response.text
            print(f"âŒ ERPNext error: {error_text[:200]}")
            
            # Try to parse error
            try:
                error_json = response.json()
                error_msg = error_json.get('message') or error_json.get('error') or f'ERPNext error: {response.status_code}'
            except:
                error_msg = f'ERPNext error: {response.status_code}'
            
            return jsonify({
                'success': False,
                'error': error_msg
            }), response.status_code
            
    except requests.exceptions.Timeout:
        return jsonify({'success': False, 'error': 'Request timeout'}), 504
    except requests.exceptions.ConnectionError:
        return jsonify({'success': False, 'error': 'Cannot connect to ERPNext'}), 503
    except Exception as e:
        print(f"âŒ Error in leave application: {str(e)}")
        return jsonify({'success': False, 'error': str(e)}), 500

@login_required
def proxy_save_technical():
    """Proxy endpoint for saving technical scores"""
    try:
        user_data = session.get('user_data')
        if not user_data:
            return jsonify({'success': False, 'error': 'Not authenticated'}), 401
            
        token = user_data.get('token', '')
        data = request.get_json()
        
        if not data:
            return jsonify({'success': False, 'error': 'No data provided'}), 400
        
        # =====================================
        # SET DATE TO LAST DAY OF SELECTED MONTH
        # =====================================
        from datetime import datetime
        import calendar
        
        selected_month = data.get('month')
        current_year = datetime.now().year
        
        # If selected month is previous month and it's January, year should be previous year
        month_names = [
            'January', 'February', 'March', 'April', 'May', 'June',
            'July', 'August', 'September', 'October', 'November', 'December'
        ]
        
        current_month_num = datetime.now().month
        selected_month_num = month_names.index(selected_month) + 1 if selected_month in month_names else current_month_num
        
        # Determine correct year
        if selected_month_num > current_month_num:
            # Selected month is in previous year (e.g., in January, December is previous year)
            year = current_year - 1
        else:
            year = current_year
        
        # Get last day of the month
        last_day = calendar.monthrange(year, selected_month_num)[1]
        
        # Set posting date to last day of selected month
        posting_date = f"{year}-{selected_month_num:02d}-{last_day:02d}"
        
        # Add to data
        data['posting_date'] = posting_date
        
        print(f"ðŸ“ Saving technical scores for {data.get('employee')} - {selected_month} {year}")
        print(f"ðŸ“… Posting date set to: {posting_date}")
        
        url = "https://erp.wttint.com/api/method/wtt_module.customization.custom.mobile_web.save_technical"
        
        headers = {
            'Authorization': token,
            'Content-Type': 'application/json'
