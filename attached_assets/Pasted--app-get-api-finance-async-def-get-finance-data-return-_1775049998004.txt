@app.get("/api/finance/kpis")
async def get_finance_kpis(project: str = "WTT-0528"):
    url = "https://erp.wttint.com/api/method/wtt_module.customization.custom.rfq.get_project_financials"
    params = {"project": project}
    headers = get_auth_headers()
    try:
        async with httpx.AsyncClient(timeout=60.0) as client:
            response = await client.get(url, params=params, headers=headers)
            if response.status_code != 200:
                print(f"Finance API returned status code: {response.status_code}")
                return {"error": "API request failed"}
            
            data = response.json()
            if data.get("message"):
                # Map API response to our KPI structure
                message = data["message"]
                return {
                    "project_budget": 0,  # Not provided in API
                    "po_cost": message.get("po_cost", 0),
                    "pr_cost": message.get("pr_cost", 0),
                    "other_expenses": (
                        message.get("cash_request", 0) + 
                        message.get("request_for_payment", 0) + 
                        message.get("ticket_booking", 0) + 
                        message.get("operational_loss", 0)
                    ),
                    "extra_expenses": message.get("operational_loss", 0),
                    "salary": message.get("salary", 0),
                    "cash_request": message.get("cash_request", 0),
                    "req_payment": message.get("request_for_payment", 0),
                    "ticket_booking": message.get("ticket_booking", 0),
                    "petty_cash": 0,  # Not provided in API
                    "claim": message.get("claim", 0),
                    "advance": message.get("advance", 0)
                }
            return {"error": "No data in response"}
    except Exception as e:
        print(f"Error fetching finance KPIs: {str(e)}")
        return {"error": str(e)}

@app.get("/api/finance/cash_request")
async def get_finance_cash_request(project: str = "WTT-0528", from_date: str = None, to_date: str = None):
    url = "https://erp.wttint.com/api/method/wtt_module.customization.custom.rfq.get_cash_request"
    params = {"project": project}
    if from_date:
        params["from_date"] = from_date
    if to_date:
        params["to_date"] = to_date
    
    print(f"=== CASH REQUEST API DEBUG ===")
    print(f"URL: {url}")
    print(f"Params: {params}")
    
    headers = get_auth_headers()
    try:
        async with httpx.AsyncClient(timeout=60.0) as client:
            response = await client.get(url, params=params, headers=headers)
            print(f"Response Status: {response.status_code}")
            
            if response.status_code != 200:
                print(f"Error Response Text: {response.text}")
                return []
            
            data = response.json()
            print(f"Parsed JSON: {data}")
            
            message = data.get("message", [])
            print(f"Message field: {message}")
            print(f"Message type: {type(message)}")
            print(f"Message length: {len(message) if isinstance(message, list) else 'Not a list'}")
            
            if message and len(message) > 0:
                print(f"First cash request item: {message[0]}")
                print(f"First cash request item keys: {list(message[0].keys()) if isinstance(message[0], dict) else 'Not a dict'}")
            
            # Map API response to frontend expected format with comprehensive field mapping
            result = []
            for item in message:
                print(f"Processing cash request item: {item}")
                
                # Try multiple possible field names for each field
                date_value = (
                    item.get("transaction_date") or 
                    item.get("date") or 
                    item.get("posting_date") or
                    item.get("creation") or
                    ""
                )
                
                entry_no_value = (
                    item.get("name") or 
                    item.get("entry_no") or 
                    item.get("document_no") or
                    ""
                )
                
                remarks_value = (
                    item.get("remarks") or 
                    item.get("description") or 
                    item.get("purpose") or
                    item.get("cash_purpose") or
                    ""
                )
                
                created_by_value = (
                    item.get("created_by") or 
                    item.get("owner") or 
                    item.get("employee") or
                    item.get("employee_name") or
                    "System"
                )
                
                amount_value = (
                    item.get("total") or 
                    item.get("amount") or 
                    item.get("grand_total") or
                    item.get("net_total") or
                    0
                )
                
                approved_by_value = (
                    item.get("workflow_state") or 
                    item.get("approved_by") or 
                    item.get("status") or
                    item.get("docstatus") or
                    ""
                )
                
                # Convert amount to number if it's a string
                if isinstance(amount_value, str):
                    try:
                        amount_value = float(amount_value)
                    except (ValueError, TypeError):
                        amount_value = 0
                
                result.append({
                    "date": date_value,
                    "entry_no": entry_no_value,
                    "remarks": remarks_value,
                    "created_by": created_by_value,
                    "amount": amount_value,
                    "approved_by": approved_by_value
                })
                
                print(f"Mapped cash request item: {result[-1]}")
            
            print(f"Final mapped result: {result}")
            print(f"=== END CASH REQUEST DEBUG ===")
            return result
            
    except Exception as e:
        print(f"Exception in cash_request: {str(e)}")
        return []
