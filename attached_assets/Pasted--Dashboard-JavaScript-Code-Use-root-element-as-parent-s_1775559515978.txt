(function() {
    // Function to get current time greeting
    function getTimeGreeting() {
        const hour = new Date().getHours();
        if (hour < 12) return "Good Morning";
        else if (hour < 17) return "Good Afternoon";
        else return "Good Evening";
    }

    // Function to format time (HH:MM:SS to HH:MM AM/PM)
    function formatTime(timeString) {
        if (!timeString) return '--';
        const [hours, minutes] = timeString.split(':');
        const hour = parseInt(hours);
        const ampm = hour >= 12 ? 'PM' : 'AM';
        const hour12 = hour % 12 || 12;
        return `${hour12}:${minutes} ${ampm}`;
    }

    // Function to calculate total incidents
    function calculateTotalIncidents(positive, negative) {
        return (positive || 0) + (negative || 0);
    }

    // Function to calculate total reminders and categorize them
    function processReminders(reminders) {
        if (!reminders || !Array.isArray(reminders)) {
            return { total: 0, today: 0, upcoming: 0, overdue: 0 };
        }

        const today = new Date().toISOString().split('T')[0];
        let todayCount = 0;
        let upcomingCount = 0;
        let overdueCount = 0;

        reminders.forEach(reminder => {
            const reminderDate = reminder.reminder_date;
            if (reminderDate === today) {
                todayCount++;
            } else if (reminderDate > today) {
                upcomingCount++;
            } else {
                overdueCount++;
            }
        });

        return {
            total: reminders.length,
            today: todayCount,
            upcoming: upcomingCount,
            overdue: overdueCount
        };
    }

// Global variable to store employee value

// Function to get employee data and initialize everything
function initializePerformanceDashboard() {
    frappe.call({
        method: 'wtt_module.customization.custom.custom_dashboard.get_employee_by_user',
        args: {
            user_id: frappe.session.user
        },
        callback: function(response) {
            if (response.message) {
                emp_value = response.message; // Store employee value globally
                console.log('Employee found:', emp_value);
                
                // Now that we have employee data, set up the dashboard
                setDefaultDateFiltersAndLoad();
                addAutoFilterListeners();
            } else {
                console.error('No employee found for the current user');
                frappe.msgprint('No employee record found for your user account.');
            }
        },
        error: function(error) {
            console.error('Error fetching employee data:', error);
            frappe.msgprint('Error loading employee data. Please try again.');
        }
    });
}

function fetchperformanceData(fromDate, toDate) {
    // Check if emp_value is available
    if (!emp_value) {
        console.error('emp_value is not available');
        // frappe.msgprint('Employee data not loaded. Please refresh the page.');
        return;
    }

    console.log('Fetching data for employee:', emp_value, 'from:', fromDate, 'to:', toDate);
    
    frappe.call({
        method: 'wtt_module.customization.custom.custom_dashboard.get_performance_data',
        args: {
            employee: emp_value,
            from_date: fromDate,
            to_date: toDate
        },
        callback: function (response) {
            console.log('Performance data response:', response);
            
            if (response.message && response.message.length > 0) {
                const item = response.message[0];
                console.log('Performance data item:', item);
                
                const safePercent = (val) =>
                    (val !== null && val !== undefined) ? parseFloat(val).toFixed(2) + "%" : "0.00%";
                
                // Update UI elements
                const tasksPointEl = root_element.querySelector('#tasks_point');
                const taskCountEl = root_element.querySelector('#task_count');
                const incidentCountEl = root_element.querySelector('#incident_count');
                const techCountEl = root_element.querySelector('#tech_count');
                const behavCountEl = root_element.querySelector('#behav_count');
                const repCountEl = root_element.querySelector('#rep_count');
                
                if (tasksPointEl) tasksPointEl.textContent = safePercent(item.total_avg);
                if (taskCountEl) taskCountEl.textContent = safePercent(item.task_allocation);
                if (incidentCountEl) incidentCountEl.textContent = safePercent(item.incident_score);
                if (techCountEl) techCountEl.textContent = safePercent(item.technical_criteria_percentage);
                if (behavCountEl) behavCountEl.textContent = safePercent(item.behavioral_criteria_percentage);
                 if (repCountEl) repCountEl.textContent = safePercent(item.report_percentage);
                
                // Log debug info if available
                if (item.debug_info) {
                    console.log('Debug info:', item.debug_info);
                }
            } else {
                console.log('No performance data found');
                frappe.msgprint("No performance data found for the selected date range.");
                
                // Reset UI to show 0.00%
                const elements = ['#tasks_point', '#task_count', '#incident_count', '#tech_count', '#behav_count'];
                elements.forEach(selector => {
                    const el = root_element.querySelector(selector);
                    if (el) el.textContent = "0.00%";
                });
            }
        },
        error: function(error) {
            console.error('Error fetching performance data:', error);
            frappe.msgprint('Error loading performance data. Please try again.');
        }
    });
}

function setDefaultDateFiltersAndLoad() {
    const startDateInput = root_element.querySelector('#start-date');
    const endDateInput = root_element.querySelector('#end-date');
    
    if (!startDateInput || !endDateInput) {
        console.error('Date input elements not found');
        return;
    }
    
    // Get current date
    const today = new Date();
    
    // Calculate first day of current month
    const firstDay = new Date(today.getFullYear(), today.getMonth(), 1);
    
    // Calculate last day of current month
    const lastDay = new Date(today.getFullYear(), today.getMonth() + 1, 0);
    
    // Format dates to YYYY-MM-DD format
    const formatDate = (date) => {
        const year = date.getFullYear();
        const month = String(date.getMonth() + 1).padStart(2, '0');
        const day = String(date.getDate()).padStart(2, '0');
        return `${year}-${month}-${day}`;
    };
    
    const fromDate = formatDate(firstDay);
    const toDate = formatDate(lastDay);
    
    // Set the input values
    startDateInput.value = fromDate;
    endDateInput.value = toDate;
    
    console.log(`Default dates set: ${fromDate} to ${toDate}`);
    console.log(`Current month: ${today.toLocaleString('default', { month: 'long', year: 'numeric' })}`);
    
    // Trigger data fetch functions
    try {
        
        fetchhalfData(fromDate, toDate);
        fetchperformanceData(fromDate, toDate);
        fetchincidentData(fromDate, toDate);
        fetchlateData(fromDate, toDate);
        fetchattendData(fromDate, toDate);
        fetchpresentData(fromDate, toDate);    
       
       
    } catch (error) {
        console.error('Error fetching data:', error);
    }
}

// Alternative version with more flexibility
function setDateFiltersAndLoad(monthOffset = 0) {
    const startDateInput = root_element.querySelector('#start-date');
    const endDateInput = root_element.querySelector('#end-date');
    
    if (!startDateInput || !endDateInput) {
        console.error('Date input elements not found');
        return;
    }
    
    const today = new Date();
    const targetDate = new Date(today.getFullYear(), today.getMonth() + monthOffset, 1);
    
    const firstDay = new Date(targetDate.getFullYear(), targetDate.getMonth(), 1);
    const lastDay = new Date(targetDate.getFullYear(), targetDate.getMonth() + 1, 0);
    
    const formatDate = (date) => {
        return date.toISOString().split('T')[0];
    };
    
    const fromDate = formatDate(firstDay);
    const toDate = formatDate(lastDay);
    
    startDateInput.value = fromDate;
    endDateInput.value = toDate;
    
    const monthName = targetDate.toLocaleString('default', { month: 'long', year: 'numeric' });
    console.log(`Dates set for ${monthName}: ${fromDate} to ${toDate}`);
    
    // Batch fetch all data
    const fetchFunctions = [
        
        () => fetchhalfData(fromDate, toDate),
        () => fetchperformanceData(fromDate, toDate),
        () => fetchincidentData(fromDate, toDate),
        () => fetchlateData(fromDate, toDate),
        () => fetchattendData(fromDate, toDate),
        () => fetchpresentData(fromDate, toDate),
      
        
    ];
    
    fetchFunctions.forEach((fetchFn, index) => {
        try {
            fetchFn();
        } catch (error) {
            console.error(`Error in fetch function ${index + 1}:`, error);
        }
    });
}

function addAutoFilterListeners() {
    const startDateInput = root_element.querySelector('#start-date');
    const endDateInput = root_element.querySelector('#end-date');
    
    if (!startDateInput || !endDateInput) {
        console.error('Date input elements not found for listeners');
        return;
    }
    
    const updateDataOnChange = () => {
        const fromDate = startDateInput.value;
        const toDate = endDateInput.value;
        
        console.log('Date changed - from:', fromDate, 'to:', toDate);
        
        if (fromDate && toDate) {
           
            fetchhalfData(fromDate, toDate);
            fetchperformanceData(fromDate, toDate);
            fetchincidentData(fromDate, toDate);
            fetchlateData(fromDate, toDate);
            fetchattendData(fromDate, toDate);
            fetchpresentData(fromDate, toDate);
           
            
        } else {
            console.warn('One or both dates are missing');
        }
    };
    
    startDateInput.addEventListener('change', updateDataOnChange);
    endDateInput.addEventListener('change', updateDataOnChange);
    
    console.log('Date change listeners added');

function addAutoFilterListeners() {
    const startDateInput = root_element.querySelector('#start-date');
    const endDateInput = root_element.querySelector('#end-date');

    const updateDataOnChange = () => {
        const fromDate = startDateInput.value;
        const toDate = endDateInput.value;

        if (fromDate && toDate) {
           
            fetchhalfData(fromDate, toDate);
            fetchperformanceData(fromDate, toDate);
            fetchincidentData(fromDate, toDate);
            fetchlateData(fromDate, toDate);
            fetchattendData(fromDate, toDate);
             fetchpresentData(fromDate, toDate);
           
        }
    };

    startDateInput.addEventListener('change', updateDataOnChange);
    endDateInput.addEventListener('change', updateDataOnChange);
}
