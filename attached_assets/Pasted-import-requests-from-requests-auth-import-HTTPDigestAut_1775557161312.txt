# ================= HELPERS =================

def clean_digits(num):
    return re.sub(r"\D", "", str(num or ""))

def get_phone(num):
    digits = clean_digits(num)
    if len(digits) >= 10:
        return digits[-10:]
    return None

def get_extension(num):
    digits = clean_digits(num)
    if 1 <= len(digits) <= 4:
        return digits
    return None

# Defined OUTSIDE the loop â€” normalizes time format differences from ERP
def normalize_time(t):
    return str(t or "").replace("T", " ").strip()[:19]

# ================= AUDIO ENHANCEMENT =================

def enhance_audio(input_file):

    enhanced = "clean_" + input_file

    try:
        result = subprocess.run([
            "ffmpeg",
            "-y",
            "-i", input_file,
            "-ac", "1",
            "-ar", "16000",
            "-af", "loudnorm,highpass=f=200,lowpass=f=3000",
            enhanced
        ], stdout=subprocess.PIPE, stderr=subprocess.PIPE)

        if os.path.exists(enhanced) and os.path.getsize(enhanced) > 1000:
            return enhanced
        else:
            print(f"  Enhancement failed, using original.")
            return input_file

    except FileNotFoundError:
        print("  ffmpeg not found â€” using original file.")
        return input_file

    except Exception as e:
        print(f"  Audio enhancement error: {e}")
        return input_file

# ================= TRANSCRIPTION =================

def transcribe_audio(file_path):

    try:
        print(f"  Transcribing: {file_path}")

        enhanced_file = enhance_audio(file_path)

        with open(enhanced_file, "rb") as audio:

            result = client.audio.transcriptions.create(
                model="gpt-4o-mini-transcribe",
                file=audio,
                prompt="""
Conversation may contain Tamil, English or Hindi.
Multiple speakers may exist.
Transcribe exactly what is spoken.
Do not translate.
"""
            )

        raw_text = result.text.strip()

        print(f"  Raw transcript ({len(raw_text)} chars): {raw_text[:200]}")

        if not raw_text:
            return ""

        # Clean up enhanced file
        if enhanced_file != file_path and os.path.exists(enhanced_file):
            os.remove(enhanced_file)

        # Speaker separation + cleanup
        response = client.chat.completions.create(
            model="gpt-4o-mini",
            temperature=0.1,
            messages=[
                {
                    "role": "system",
                    "content": """
Clean and structure transcript.

Rules:

Language may be:
Tamil
English
Hindi
Mixed sentences allowed.

Detect speakers.

Return format:

Speaker 1:
text

Speaker 2:
text

Speaker 3:
text
"""
                },
                {"role": "user", "content": raw_text}
            ]
        )

        final = response.choices[0].message.content.strip()

        print(f"  Final transcript: {final[:200]}")

        return final

    except Exception as e:
        print(f"  Transcript error: {e}")
        return ""

# ================= CHECK SKIP LIST =================

def skip_number(phone):

    try:
        r = requests.get(
            f"{ERP_BASE_URL}/api/resource/Skip Call Logs",
            headers=ERP_HEADERS,
            params={
                "filters": json.dumps([["phone_number", "=", phone]])
            }
        )
        return bool(r.json().get("data"))

    except Exception as e:
        print(f"  Skip check error: {e}")
        return False

# ================= RECRUITMENT TRACKER =================

def create_recruitment_tracker(phone):

    # Check skip list first
    if skip_number(phone):
        print(f"  Skipped â€” {phone} is in skip list.")
        return

    # Check if tracker already exists for this number
    existing = requests.get(
        f"{ERP_BASE_URL}/api/resource/Recruitment Tracker",
        headers=ERP_HEADERS,
        params={
            "filters": json.dumps([["contact", "=", phone]])
        }
    )

    if existing.json().get("data"):
        print(f"  Recruitment Tracker already exists for: {phone}")
        return

    # Create new tracker
    r = requests.post(
        f"{ERP_BASE_URL}/api/resource/Recruitment Tracker",
        headers=ERP_HEADERS,
        json={
            "doctype": "Recruitment Tracker",
            "contact": phone
        }
    )

    if r.status_code == 200:
        print(f"  Recruitment Tracker created for: {phone}")
    else:
        print(f"  Recruitment Tracker failed: {r.status_code} - {r.text}")

# ================= FETCH & PROCESS CALLS =================

def fetch_and_process(start_time_manual=None):

    today = datetime.now().strftime("%Y-%m-%d")
    current_time = datetime.now().strftime("%Y-%m-%d %H:%M:%S")

    print("\n" + "=" * 60)
    print(f"  Fetching HR calls for : {today}")
    print(f"  Run time              : {current_time}")
    print("=" * 60)

    cdr_url = f"https://{UCM_IP}:8443/cdrapi"
    start_time_str = f"2026-03-26T00:00:00"
    
    params = {
        "format"    : "JSON",
        "startTime" : start_time_str,
        "endTime"   : f"{today}T{current_time.split(' ')[1]}",
        "numRecords": 1000  # Increased batch size, will loop if needed
    }

    # Loop to get all records
    all_cdr_data = []
    offset = 0
    batch_size = 1000
    
    while True:
        params["numRecords"] = batch_size
        
        try:
            cdr_response = requests.get(
                cdr_url,
                params=params,
                auth=HTTPDigestAuth(UCM_USER, UCM_PASS),
                verify=False,
                timeout=30
            )
            batch_data = cdr_response.json().get("cdr_root", [])
            
            if not batch_data:
                break  # No more records
                
            all_cdr_data.extend(batch_data)
            print(f"  Fetched batch: {len(batch_data)} records (Total: {len(all_cdr_data)})")
            
            # If we got less than requested, we're done
            if len(batch_data) < batch_size:
                break
                
            # Move to next batch (adjust start time if API doesn't support offset)
            # For now, we'll assume the API returns all available records
            break
            
        except Exception as e:
            print(f"  CDR fetch failed: {e}")
            break
    
    cdr_data        = [c for c in all_cdr_data if c.get("start")]
    cdr_data_sorted = sorted(
        cdr_data,
        key=lambda x: datetime.strptime(x["start"], "%Y-%m-%d %H:%M:%S"),
        reverse=True
    )

    print(f"  Calls Found: {len(cdr_data_sorted)}")

    # ================= PROCESS EACH CALL =================

    new_calls     = 0
    skipped_calls = 0

    for call in cdr_data_sorted:

        src        = call.get("src")
        dst        = call.get("dst")
        start_time = call.get("start")
        recordfile = call.get("recordfiles") or call.get("recordfile")

        src_phone = get_phone(src)
        dst_phone = get_phone(dst)
        src_ext   = get_extension(src)
        dst_ext   = get_extension(dst)

        if src_ext in HR_EXTENSIONS:
            phone_number = dst_phone
            extension    = src_ext
            call_type    = "Outgoing"

        elif dst_ext in HR_EXTENSIONS:
            phone_number = src_phone
            extension    = dst_ext
            call_type    = "Incoming"

        else:
            continue

        if not phone_number:
            continue

        owner_name  = EXTENSION_OWNER.get(extension, extension)
        parent_name = f"{phone_number}"
        call_date   = start_time.split(" ")[0]

        # ===== STEP 1: Create or get parent doc in ERP =====

        parent_check = requests.get(
            f"{ERP_BASE_URL}/api/resource/{PARENT_DOCTYPE}/{parent_name}",
            headers=ERP_HEADERS
        )

        if parent_check.status_code != 200:
            requests.post(
                f"{ERP_BASE_URL}/api/resource/{PARENT_DOCTYPE}",
                json={
                    "doctype"     : PARENT_DOCTYPE,
                    "name"        : parent_name,
                    "phone_number": phone_number
                },
                headers=ERP_HEADERS
            )

        parent_doc_response = requests.get(
            f"{ERP_BASE_URL}/api/resource/{PARENT_DOCTYPE}/{parent_name}",
            headers=ERP_HEADERS
        )

        if parent_doc_response.status_code != 200:
            print(f"  Failed to fetch parent doc: {parent_doc_response.status_code}")
            continue

        parent_doc_json = parent_doc_response.json()

        if "data" not in parent_doc_json:
            print(f"  Unexpected ERP response for {parent_name}")
            continue

        parent_doc = parent_doc_json["data"]
        rows       = parent_doc.get(CHILD_FIELD, [])

        # ===== Duplicate check â€” by call_time only (normalized) =====
        saved_times  = set(normalize_time(r.get("call_time")) for r in rows)
        current_time = normalize_time(start_time)

        if current_time in saved_times:
            skipped_calls += 1
            continue  # already saved â€” skip silently

        # New call â€” process it
        new_calls += 1
        print(f"\n  New call: {call_type} | {phone_number} | {start_time} | {owner_name}")

        # ===== STEP 2: Download recording + transcribe =====

        recording_url   = ""
        transcript_text = ""
        summary_text    = ""

        if recordfile:

            rec_url  = f"https://{UCM_IP}:8443/recapi"
            filename = recordfile.split("/")[-1].replace("@", "")
            dt       = datetime.strptime(start_time, "%Y-%m-%d %H:%M:%S")

            folders = [
                f"monitor/{dt.strftime('%Y-%m-%d')}",
                f"monitor/{dt.strftime('%Y-%m')}",
                f"monitor/{dt.strftime('%Y')}/{dt.strftime('%m')}",
                "monitor"
            ]

            rec_response = None

            for fd in folders:
                r = requests.get(
                    rec_url,
                    params={"filename": filename, "filedir": fd},
                    auth=HTTPDigestAuth(UCM_USER, UCM_PASS),
                    verify=False
                )
                if r.status_code == 200 and len(r.content) > 10000:
                    rec_response = r
                    print(f"  Recording found: {fd} ({len(r.content)} bytes)")
                    break

            if rec_response:

                with open(filename, "wb") as f:
                    f.write(rec_response.content)

                transcript_text = transcribe_audio(filename)
                summary_text    = analyze_call(transcript_text)

                # Upload recording to ERP
                with open(filename, "rb") as f:
                    upload = requests.post(
                        f"{ERP_BASE_URL}/api/method/upload_file",
                        headers=ERP_HEADERS,
                        files={
                            "file"      : (filename, f),
                            "doctype"   : (None, PARENT_DOCTYPE),
                            "docname"   : (None, parent_name),
                            "is_private": (None, "1")
                        }
                    )

                if upload.status_code == 200:
                    recording_url = upload.json()["message"]["file_url"]
                    print(f"  Recording uploaded: {recording_url}")
                else:
                    print(f"  Recording upload failed: {upload.status_code}")

                if os.path.exists(filename):
                    os.remove(filename)

            else:
                print(f"  Recording not found in any folder: {filename}")

        else:
            print("  No recording for this call.")

        # ===== STEP 3: Save to ERP child table =====

        rows.append({
            "call_date"      : call_date,
            "phone_number"   : phone_number,
            "extension"      : extension,
            "extension_owner": owner_name,
            "call_type"      : call_type,
            "call_time"      : start_time,
            "recording"      : recording_url,
            "call_transcript": transcript_text,
            "summary"        : summary_text
        })

        update = requests.put(
            f"{ERP_BASE_URL}/api/resource/{PARENT_DOCTYPE}/{parent_name}",
            json={CHILD_FIELD: rows},
            headers=ERP_HEADERS
        )

        print(f"  ERP save status: {update.status_code}")

        # ===== STEP 4: Recruitment Tracker â€” Incoming calls only =====

        if call_type == "Incoming":
            create_recruitment_tracker(phone_number)
        else:
            print("  Outgoing call â€” skipping recruitment tracker.")

        print("-" * 50)

    # Summary after each run
    print(f"\n  Run complete â€” New: {new_calls} | Already saved: {skipped_calls}")


# ================= MAIN LOOP â€” Every 10 Minutes =================

while True:
